## 数据库建表（可选）

手动执行以下sql：  

```
-- 风险事件表
CREATE TABLE IF NOT EXISTS zenvis.risk_event (
    id String NOT NULL COMMENT '唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    asset_id String COMMENT '资产ID',
    type String COMMENT '威胁类型',
    detail String COMMENT '威胁详情',
    label Array(String) COMMENT '标签',
    risk_level String COMMENT '风险等级',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 数据风险
CREATE TABLE IF NOT EXISTS zenvis.risk_data (
    id String NOT NULL COMMENT '唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    asset_id String COMMENT '资产ID',
    type String COMMENT '威胁类型',
    detail String COMMENT '威胁详情',
    verification_method String COMMENT '验证方法',
    verification_result String COMMENT '验证结果',
    label Array(String) COMMENT '标签',
    risk_level String COMMENT '风险等级',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 漏洞风险表
CREATE TABLE IF NOT EXISTS zenvis.risk_vulnerability (
    id String NOT NULL COMMENT '漏洞风险唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    asset_id String COMMENT '资产ID',
    vulnerability_name String COMMENT '漏洞名称',
    vulnerability_id String COMMENT '漏洞编号',
    vulnerability_category String COMMENT '漏洞分类',
    vulnerability_description String COMMENT '漏洞描述',
    cvss_score Float64 COMMENT 'CVSS评分',
    cvss_vector String COMMENT 'CVSS向量',
    affected_scope String COMMENT '影响范围',
    affected_component String COMMENT '影响组件',
    affected_version String COMMENT '影响版本',
    fixed_version String COMMENT '修复版本',
    fix_suggestion String COMMENT '修复建议',
    vulnerability_status String COMMENT '漏洞状态',
    verification_method String COMMENT '验证方法',
    verification_result String COMMENT '验证结果',
    label Array(String) COMMENT '标签',
    risk_level String COMMENT '风险等级',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 基线风险表
CREATE TABLE IF NOT EXISTS zenvis.risk_baseline (
    id String NOT NULL COMMENT '唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    asset_id String COMMENT '资产ID',
    configuration_name String COMMENT '配置项名称',
    expected_value String COMMENT '预期配置值',
    current_value String COMMENT '当前配置值',
    verification_method String COMMENT '验证方法',
    verification_result String COMMENT '验证结果',
    label Array(String) COMMENT '标签',
    risk_level String COMMENT '风险等级',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 弱口令风险表
CREATE TABLE IF NOT EXISTS zenvis.risk_weak (
    id String NOT NULL COMMENT '唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    asset_id String COMMENT '资产ID',
    username String COMMENT '使用该弱口令的用户名',
    password String COMMENT '弱口令',
    password_type String COMMENT '弱口令类型',
    verification_method String COMMENT '验证方法',
    verification_result String COMMENT '验证结果',
    label Array(String) COMMENT '标签',
    risk_level String COMMENT '风险等级',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 攻击风险表
CREATE TABLE IF NOT EXISTS zenvis.risk_attack (
    id String NOT NULL COMMENT '唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    asset_id String COMMENT '资产ID',
    type String COMMENT '威胁类型',
    detail String COMMENT '威胁详情',
    verification_method String COMMENT '验证方法',
    verification_result String COMMENT '验证结果',
    label Array(String) COMMENT '标签',
    risk_level String COMMENT '风险等级',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);
```
