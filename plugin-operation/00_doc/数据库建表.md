## 数据库建表（可选）

手动执行以下sql：  

```
-- 启动事件
CREATE TABLE IF NOT EXISTS zenvis.operation_start_event (
    id String NOT NULL COMMENT '事件唯一编号',
    user_id String COMMENT '用户ID',
    device_id String COMMENT '设备ID',
    device_model String COMMENT '设备型号',
    device_os String COMMENT '设备系统',
    app_id String COMMENT '应用ID',
    app_name String COMMENT '应用名称',
    package_name String COMMENT '应用包名',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 点击事件
CREATE TABLE IF NOT EXISTS zenvis.operation_click_event (
    id String NOT NULL COMMENT '事件唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    click_x Float64 COMMENT '点击坐标x',
    click_y Float64 COMMENT '点击坐标y',
    component_type String COMMENT '点击组件类型',
    component_name String COMMENT '组件名称',
    page_path String COMMENT '所属页面路径',
    page_name String COMMENT '所属页面名称',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 页面访问事件
CREATE TABLE IF NOT EXISTS zenvis.operation_page_event (
    id String NOT NULL COMMENT '事件唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    page_path String COMMENT '页面路径',
    page_name String COMMENT '页面名称',
    referrer String COMMENT '跳转来源',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);


-- 函数调用事件
CREATE TABLE IF NOT EXISTS zenvis.operation_api_call_event (
    id String NOT NULL COMMENT '事件唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    caller String COMMENT '调用对象',
    callee String COMMENT '被调用对象',
    function_name String COMMENT '函数名',
    function_params String COMMENT '函数传参',
    function_return String COMMENT '函数返回值',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 网络请求事件
CREATE TABLE IF NOT EXISTS zenvis.operation_network_event (
    id String NOT NULL COMMENT '事件唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    source_ip String COMMENT '源ip',
    source_port String COMMENT '源端口',
    target_ip String COMMENT '目标ip',
    target_port String COMMENT '目标端口',
    protocol String COMMENT '协议',
    request_time DateTime64(3) COMMENT '请求时间',
    response_time DateTime64(3) COMMENT '响应时间',
    request_content String COMMENT '请求内容',
    response_content String COMMENT '响应内容',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 用户拓展事件
CREATE TABLE IF NOT EXISTS zenvis.operation_extend_event (
    id String NOT NULL COMMENT '事件唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    extend String COMMENT '扩展消息',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);

-- 资源利用状态
CREATE TABLE IF NOT EXISTS zenvis.operation_performance_event (
    id String NOT NULL COMMENT '事件唯一标识',
    user_id String COMMENT '用户ID',
    start_id String COMMENT '启动ID',
    cpu_usage Float64 COMMENT 'CPU总利用率',
    cpu_user Float64 COMMENT 'CPU用户态利用率',
    cpu_system Float64 COMMENT 'CPU系统态利用率',
    cpu_idle Float64 COMMENT 'CPU空闲率',
    memory_total Int64 COMMENT '物理内存总量(KB)',
    memory_used Int64 COMMENT '物理内存使用量(KB)',
    memory_usage Float64 COMMENT '物理内存利用率',
    virtual_memory_total Int64 COMMENT '虚拟内存总量(KB)',
    virtual_memory_used Int64 COMMENT '虚拟内存使用量(KB)',
    swap_total Int64 COMMENT '交换空间总量(KB)',
    swap_used Int64 COMMENT '交换空间使用量(KB)',
    disk_read_speed Float64 COMMENT '磁盘读取速度(KB/s)',
    disk_write_speed Float64 COMMENT '磁盘写入速度(KB/s)',
    disk_throughput Float64 COMMENT '磁盘总吞吐量(KB/s)',
    disk_iops Int32 COMMENT '磁盘IOPS',
    network_send_rate Float64 COMMENT '网络发送速率(KB/s)',
    network_receive_rate Float64 COMMENT '网络接收速率(KB/s)',
    network_packet_loss Float64 COMMENT '网络丢包率',
    network_latency Float64 COMMENT '网络延迟(ms)',
    longitude Float64 COMMENT '经度',
    latitude Float64 COMMENT '纬度',
    country String COMMENT '国家',
    province String COMMENT '省份',
    city String COMMENT '城市',
    county String COMMENT '区县',
    net_type String COMMENT '网络类型',
    lan_ip String COMMENT '局域网IP',
    wan_ip String COMMENT '因特网IP',
    update_time DateTime64(3) COMMENT '更新时间',
    insert_time DateTime64(3) COMMENT '创建时间',
    event_time DateTime64(3) COMMENT '事件发生时间',
    PRIMARY KEY (id)
) ENGINE = MergeTree()
ORDER BY (id);
```
